```typescript
try {
  await db.insert(categories).values(newCategories);
```

```typescript
// Use onConflictDoNothing() to handle concurrent category initialization gracefully.
// The uniqueIndex on (userId, slug) prevents duplicates; onConflictDoNothing()
// silently skips inserts that would violate the constraint without throwing.
await db.insert(categories).values(newCategories).onConflictDoNothing();

// Re-query after insert to get the actual database state (handles concurrent init
// where another request may have inserted between our SELECT and INSERT).
const initializedCategories = await db
  .select()
  .from(categories)
  .where(eq(categories.userId, userId));
```
