export class RateLimitGuard implements CanActivate {
  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest<Request & { user?: { id: string } }>();
    const authHeader = request.headers.authorization;
    const identifier = request.user?.id || (authHeader ? authHeader.replace('Bearer ', '') : null);
    const ratelimit = getRatelimitClient();
    if (!identifier) {
      const ip = (request.headers['x-forwarded-for'] as string) || request.socket.remoteAddress || 'anonymous';
      const { success } = await ratelimit.limit(`ratelimit_${ip}`);
      if (!success) { throw new HttpException('Rate limit exceeded', HttpStatus.TOO_MANY_REQUESTS); }
      return true;
    }
    // ... rate limit by identifier
  }
}

```typescript
import { plainToInstance } from 'class-transformer';
import { IsString, IsNotEmpty, validateSync } from 'class-validator';

class EnvironmentVariables {
  @IsString()
  @IsNotEmpty()
  BETTER_AUTH_SECRET!: string;

  @IsString()
  @IsNotEmpty()
  TURSO_CONNECTION_URL!: string;

  @IsString()
  @IsNotEmpty()
  TURSO_AUTH_TOKEN!: string;

  @IsString()
  @IsNotEmpty()
  GITHUB_CLIENT_ID!: string;

  @IsString()
  @IsNotEmpty()
  GITHUB_CLIENT_SECRET!: string;

  @IsString()
  @IsNotEmpty()
  OPENROUTER_API_KEY!: string;
}

function validate(config: Record<string, unknown>) {
  const validatedConfig = plainToInstance(EnvironmentVariables, config, {
    enableImplicitConversion: true,
  });
  const errors = validateSync(validatedConfig, {
    skipMissingProperties: false,
  });

  if (errors.length > 0) {
    const messages = errors.map(
      (e) =>
        `${e.property} is required but not set. Check your .env file.`,
    );
    throw new Error(`Environment validation failed:\n${messages.join('\n')}`);
  }
  return validatedConfig;
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CORS origin check | Browser-enforced. Server rejects non-allowed origins before processing request. |
| Rate limiter → auth | Guard runs before controller. Unauthenticated requests never reach LLM service. |
| Env validation → boot | Fails fast at startup. App won't boot with missing secrets — prevents silent OAuth failures. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | CORS | mitigate | Dynamic origin function checks against allowlist. null origin allowed for mobile. |
| T-01-02 | Spoofing | Rate limiter | mitigate | IP-based fallback removed. x-forwarded-for no longer trusted. Session ID required. |
| T-01-03 | Information Disclosure | Env validation | mitigate | Validation happens at boot — no secrets logged. Error messages only mention var names, not values. |
| T-01-04 | Denial of Service | Rate limiter | accept | Authenticated users still rate-limited. Unauthenticated rejected fast (401). No resource exhaustion. |
</threat_model>
