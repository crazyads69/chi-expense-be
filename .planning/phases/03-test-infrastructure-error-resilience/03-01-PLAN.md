    Create `src/db/database.module.ts` as a `@Global()` module that provides the Drizzle client via the `DRIZZLE` token.
    
    The provider factory must create the same libsql HTTP client as `src/db/client.ts`:
    ```typescript
    import { Global, Module } from '@nestjs/common';
    import { drizzle } from 'drizzle-orm/libsql';
    import { createClient } from '@libsql/client/http';
    import * as schema from './schema';
    import { DRIZZLE } from './db-token';
    import type { DrizzleDatabase } from './db-token';

    @Global()
    @Module({
      providers: [
        {
          provide: DRIZZLE,
          useFactory: (): DrizzleDatabase => {
            const client = createClient({
              url: process.env.TURSO_CONNECTION_URL || 'file:local.db',
              authToken: process.env.TURSO_AUTH_TOKEN,
            });
            return drizzle(client, { schema });
          },
        },
      ],
      exports: [DRIZZLE],
    })
    export class DatabaseModule {}
    ```
  </action>

  </verify>
