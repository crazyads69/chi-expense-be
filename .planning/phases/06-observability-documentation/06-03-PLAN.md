   ```typescript
   import { DocumentBuilder, SwaggerModule } from '@nestjs/swagger';
   
   // ... after const app = await NestFactory.create(...)
   
   const config = new DocumentBuilder()
     .setTitle('Chi Expense API')
     .setDescription('Zero-friction expense tracking API for Chi Expense mobile app')
     .setVersion('1.0.0')
     .addBearerAuth(
       {
         type: 'http',
         scheme: 'bearer',
         bearerFormat: 'JWT',
         description: 'Enter your Bearer token (session token from Better Auth)',
       },
       'bearer',
     )
     .build();
   
   const document = SwaggerModule.createDocument(app, config);
   SwaggerModule.setup('api/docs', app, document, {
     swaggerOptions: {
       persistAuthorization: true,
     },
   });
   ```

3. Ensure the Swagger setup comes AFTER security middleware (helmet, CORS) but BEFORE `app.init()`.
