```typescript
import { Injectable } from '@nestjs/common';
import { Inject } from '@nestjs/common';
import { DRIZZLE } from '../db/db-token';
import { eq } from 'drizzle-orm';
import { categories } from '../db/schema';
import { getRedisClient } from '../lib/redis';
import { Logger } from 'nestjs-pino';

const CACHE_TTL = 60; // 60 seconds
const CACHE_PREFIX = 'categories';

@Injectable()
export class CategoriesService {
  private readonly logger = new Logger(CategoriesService.name);

  constructor(@Inject(DRIZZLE) private readonly db: any) {}

  async list(userId: string) {
    const cacheKey = `${CACHE_PREFIX}:${userId}`;
    const redis = getRedisClient();

    try {
      // Try to get from cache
      const cached = await redis.get(cacheKey);
      if (cached) {
        this.logger.debug({ cache: 'hit', userId }, 'Categories cache hit');
        return JSON.parse(cached);
      }
    } catch (error) {
      this.logger.warn({ error: error.message }, 'Redis cache read failed');
    }

    // Cache miss - fetch from DB
    this.logger.debug({ cache: 'miss', userId }, 'Categories cache miss');
    const result = await this.db
      .select()
      .from(categories)
      .where(eq(categories.userId, userId));

    try {
      // Store in cache
      await redis.setex(cacheKey, CACHE_TTL, JSON.stringify(result));
    } catch (error) {
      this.logger.warn({ error: error.message }, 'Redis cache write failed');
    }

    return result;
  }
}
```
